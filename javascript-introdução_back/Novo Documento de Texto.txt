 <script>
        function obterSexoValido(pessoa) {
            let sexo;
            do {
                sexo = prompt("Digite o sexo da pessoa " + pessoa + " (1 - Masculino, 2 - Feminino):");
            } while (sexo !== "1" && sexo !== "2");
            return parseInt(sexo);
        }

        function programa() {
            let altura1 = parseFloat(prompt("Digite a altura da pessoa 1 (em metros):"));
            let sexo1 = obterSexoValido(1);

            let altura2 = parseFloat(prompt("Digite a altura da pessoa 2 (em metros):"));
            let sexo2 = obterSexoValido(2);

            let maiorAltura, sexoMaior;

            if (altura1 > altura2) {
                maiorAltura = altura1;
                sexoMaior = sexo1;
            } else {
                maiorAltura = altura2;
                sexoMaior = sexo2;
            }

            let sexoTexto = (sexoMaior === 1) ? "masculino" : "feminino";

            alert("A maior altura é: " + maiorAltura + "m\nSexo: " + sexoTexto);
        }

        // Executa ao carregar a página
        programa();
    </script>
